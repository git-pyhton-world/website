#project 4
import random
import tkinter as tk
from tkinter import messagebox

# Initialize the dictionary
dict1 = {'apple': 'fruit', 'cricket': 'sports'}
lst = list(dict1.keys())
str1 = random.choice(lst)

# Function to check the user's input
def check_guess():
    user_input = entry.get()
    if dict1[str1] == user_input:
        messagebox.showinfo("Result", "You have won!")
    else:
        messagebox.showinfo("Result", f"You have lost. The correct answer was: {dict1[str1]}")

# Create the main window
root = tk.Tk()
root.title("Nickname Guessing Game")
root.geometry("400x300")  # Set the size of the window

# Create and place widgets with more styling
header_label = tk.Label(root, text="Guess the Nickname!", font=("Arial", 20, "bold"))
header_label.pack(pady=20)

question_label = tk.Label(root, text=f"Match the word: {str1}", font=("Arial", 16))
question_label.pack(pady=10)

entry = tk.Entry(root, font=("Arial", 14), width=30)
entry.pack(pady=10)

submit_button = tk.Button(root, text="Submit", font=("Arial", 14), command=check_guess)
submit_button.pack(pady=20)

# Start the Tkinter event loop
root.mainloop()
