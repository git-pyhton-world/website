#opening and adding a comment
def friend(friend):
    global comm
    global wri
    content=comm.get()
    filename=friend
    filename+='.txt'
    wrii=wri.get()
    
    fh=open(filename,'a+')
    fh.write(wrii)
    fh.write(' : ')
    fh.write(content)
    fh.write('\n')
    fh.close()
