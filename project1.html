class Bank:
    def __init__(self, det):
        self.name = det[0]
        self.account = det[1]
        self.c_id = det[2]
        self.b_no = det[3]
        self.amount = det[4]

    def change(self, attr_name, new_value):
        if hasattr(self, attr_name):
            setattr(self, attr_name, new_value)

    def view(self):
        print(f'''Name of Account holder: {self.name}
Account Number: {self.account}
Customer Id: {self.c_id}
Branch No: {self.b_no}''')

def getvalue():
    n = input('Name:')
    ac_no = int(input('Account Number :'))
    c_id = input('Customer Id:')
    b_no = int(input('Branch Number:'))
    amount = int(input('Amount:'))
    return [n, ac_no, c_id, b_no, amount]
