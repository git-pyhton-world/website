'''to create account and changevalue of bank accounts using class , objects and user defined function'''
class Bank :

    #constructor method to store all the primary details in first
    def __init__(self,det):
        self.name=det[0]
        self.account=det[1]
        self.c_id=det[2]
        self.b_no=det[3]
        self.amount=det[4]
    
    #method to change a information
    def change(self,attr_name,new_value) :

        if hasattr(self,attr_name):
                setattr(self,attr_name,new_value)
    def view(self) :
        print(f'''Name of Account holder:{self.name}
        Account Number                  :{self.account}
        Customer Id                     :{self.c_id}
        Branch No                       :{self.b_no}''')
