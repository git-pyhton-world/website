Enter a number it will print a square with a side length of that number


#project 1
n=int(input('Enter the no. of star to be printed in one line:'))
s=0
for i in range(n,0,-1):
    for j in range(0,i):
        print('*',end=' ')
    for k in range(n,i+s):
        print('*',end=' ')
    s+=2
    print()
