Enter a number it will print a pyramid with a height length of that number


#project 2
n=int(input('Enter the number of line in the pyramid you want:'))
s=1
for i in range(n,0,-1):
    for j in range(1,i):
        print(end='  ')
    for k in range(n-s,i):
        print('*',end=' ')
    s+=3
    print()
