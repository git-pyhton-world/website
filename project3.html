# Project 3
def isoperator(x):
    if x in '+-*/' :
        return True
def posttopre(postexpression):
    stack=[]
    for i in range(len(postexpression)) :
        if isoperator(postexpression[i]) :
            a=stack.pop()
            b=stack.pop()
            temp=postexpression[i]+b+a
            stack.append(temp)
        else :
            stack.append(postexpression[i])
    ans=''
    for i in stack:
        ans+=i
    return ans
post=input('Enter postexpression:')
print('prefix notation is :',posttopre(post))
