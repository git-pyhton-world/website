import tkinter
def numcheck(n1,n2):
    b=1
    print('***check whether the number is a prime number or composite number between any two number***')
   
    for n in range(n1,n2):
        for i in range(2,n//2+1):
           if n%i==0 :
                b=0
          
        if b==0:
            print(' Number',n,'is a composite number')
            
        else:
            print('Number',n,'is a prime number.')
        
        b+=1    
        print()
win1=tkinter.Tk()
buttom=tkinter.Button(win1,command=numcheck(100, 200)).pack()
win1.mainloop()
